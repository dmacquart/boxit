function submitOrder() { // Generate order number const orderNum = '#BX-' + new Date().getFullYear() + '-' + String(Math.floor(Math.random() * 1000)).padStart(3, '0'); // Create a form to submit to FormSubmit const form = document.createElement('form'); form.method = 'POST'; form.action = 'https://formsubmit.co/info@tryboxit.com'; form.style.display = 'none'; // Add all the booking data as hidden fields const fields = { '_subject': 'New Box It Booking - ' + orderNum, '_template': 'table', '_captcha': 'false', 'Order_Number': orderNum, 'Number_of_Boxes': formData.boxes, 'Delivery_Date': formData.deliveryDate, 'Delivery_Time': formData.deliveryTime, 'Full_Name': formData.fullName, 'Email': formData.email, 'Phone': formData.phone, 'Address': formData.address, 'Unit': formData.unit || 'N/A', 'City': formData.city, 'State': formData.state, 'ZIP': formData.zip, 'Instructions': formData.instructions || 'None', 'Monthly_Total': '$' + (25 + (formData.boxes * 15)) + '.00' }; // Create hidden input for each field Object.keys(fields).forEach(key => { const input = document.createElement('input'); input.type = 'hidden'; input.name = key; input.value = fields[key]; form.appendChild(input); }); // Add form to page and submit document.body.appendChild(form); form.submit(); }
